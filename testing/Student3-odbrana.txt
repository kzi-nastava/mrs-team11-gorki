    password = "test123"
    confirmPassword = "test123"
    firstName = "E2E"
    lastName = "Passenger"
    phoneNumber = "123456789"

    profileImage = ""
} | ConvertTo-Json

Invoke-RestMethod `
    -Uri "http://localhost:8080/api/auth/register" `
    -Method POST `
    -ContentType "application/json" `
    -Body $body
